6   import java.awt.event.ActionEvent;

8   public abstract class WindowInterceptorTestCase extends InterceptionTestCase {
9     protected Trigger getShowFirstDialogTrigger() {
10      return new Trigger() {
11        public void run() throws Exception {
12          JDialog firstDialog = createDialogs();
13          logger.log("trigger");
14          firstDialog.setVisible(true);
15        }
16      };
17    }
18  
19    private JDialog createDialogs() {
20      final JFrame frame = new JFrame();
21  
22      final JDialog firstDialog = new JDialog(frame, "first dialog", true);
23      addHideButton(firstDialog, "Dispose");
24  
25      JDialog secondDialog = new JDialog(frame, "second dialog", true);
26      addShowDialogAndCloseButton(firstDialog, "OK", secondDialog);
27      addHideButton(secondDialog, "Dispose");
28  
29      JDialog thirdDialog = new JDialog(frame, "third dialog", true);
30      addHideButton(thirdDialog, "Dispose");
31      addShowDialogAndCloseButton(secondDialog, "OK", thirdDialog);
32      return firstDialog;
33    }
34  
35    protected static class ShowDialogAction extends AbstractAction {
36      private boolean modal;
37  
38      public ShowDialogAction(boolean modal) {
39        super("Run");
40        this.modal = modal;
41      }
42  
43      public void actionPerformed(ActionEvent e) {
44        JDialog dialog = new JDialog();
45        dialog.setModal(modal);
46        dialog.setTitle("MyDialog");
47        dialog.setVisible(true);
48      }
49    }
50  }
