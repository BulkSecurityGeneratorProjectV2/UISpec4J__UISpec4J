14  /**
15   * TestCase which automatically enables UISpec4J's interception mechanism.
16   * <p/>
17   * This test case is useful when the default toolkit has already been initialized
18   * before an UISpec4J test was run.
19   * </p>
20   */
21  public abstract class InterceptionTestCase extends UnitTestCase {
22    protected EventLogger logger;
23  
24    protected void setUp() throws Exception {
25      super.setUp();
26      UISpecDisplay.instance().reset();
27      logger = new EventLogger();
28      UISpec4J.setWindowInterceptionTimeLimit(300);
29    }
30  
31    protected void tearDown() throws Exception {
32      super.tearDown();
33      assertEquals(0, UISpecDisplay.instance().getHandlerCount());
34      UISpecDisplay.instance().rethrowIfNeeded();
35    }
36  
37    public static JDialog createModalDialog(String title) {
38      return new JDialog(((JFrame)null), title, true);
39    }
40  
41    protected static void checkAssertionError(WindowInterceptor interceptor,
42                                                    String errorMessage) {
43      try {
44        interceptor.run();
45        throw new AssertionFailureNotDetectedError();
46      }
47      catch (AssertionError e) {
48        assertEquals(errorMessage, e.getMessage());
49      }
50      catch (InterceptionError e) {
51        if (!e.getMessage().equals(errorMessage)) {
52          throw new InterceptionError("Unexpected error (expected '" + errorMessage + "'", e);
53        }
54      }
55    }
56  
57    public static void createAndShowModalDialog(String title) {
58      createModalDialog(title).setVisible(true);
59    }
60  
61    protected void addLoggerButton(final RootPaneContainer container, final String buttonLabel) {
62      JButton button = new JButton(buttonLabel);
63      button.addActionListener(new ActionListener() {
64        public void actionPerformed(ActionEvent e) {
65          logger.log("click").add("button", buttonLabel);
66        }
67      });
68      container.getContentPane().add(button);
69    }
70  
71    protected void addHideButton(final JDialog dialog, final String buttonLabel) {
72      JButton button = new JButton(new AbstractAction(buttonLabel) {
73        public void actionPerformed(ActionEvent e) {
74          logger.log("click").add("button", buttonLabel);
75          dialog.setVisible(false);
76        }
77      });
78      dialog.getContentPane().add(button);
79    }
80  
81    protected void addShowDialogButton(final RootPaneContainer container,
82                                       final String buttonLabel,
83                                       final JDialog dialogToShow) {
84      JButton button = new JButton(new AbstractAction(buttonLabel) {
85        public void actionPerformed(ActionEvent e) {
86          logger.log("click").add("button", buttonLabel);
87          dialogToShow.setVisible(true);
88        }
89      });
90      container.getContentPane().add(button);
91    }
92  
93    protected void addShowDialogAndCloseButton(final JDialog container,
94                                               final String buttonLabel,
95                                               final JDialog dialogToShow) {
96      JButton button = new JButton(new AbstractAction(buttonLabel) {
97        public void actionPerformed(ActionEvent e) {
98          logger.log("click").add("button", buttonLabel);
99          container.setVisible(false);
100         dialogToShow.setVisible(true);
101       }
102     });
103     container.getContentPane().add(button);
104   }
105 
106   protected String getLocalLabel(String resource) {
107     return (String) UIManager.get(resource);
108   }
109 }
