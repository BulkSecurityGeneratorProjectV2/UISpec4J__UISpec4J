9   public class MainClassAdapterTest extends InterceptionTestCase {
10    public void test() throws Exception {
11      MainClassAdapter adapter = new MainClassAdapter(MyClass.class, "a", "b");
12      Window window = adapter.getMainWindow();
13      TextBox textBox = window.getTextBox();
14      assertTrue(textBox.textEquals("[a, b]"));
15    }
16  
17    public void testReusesTheInterceptedWindowOnSubsequentCalls() throws Exception {
18      MyClass.callCount = 0;
19  
20      MainClassAdapter adapter = new MainClassAdapter(MyClass.class, "a", "b");
21      Window window1 = adapter.getMainWindow();
22      Window window2 = adapter.getMainWindow();
23      assertSame(window1, window2);
24      assertEquals(1, MyClass.callCount);
25  
26      adapter.reset();
27      Window window3 = adapter.getMainWindow();
28      assertNotSame(window1, window3);
29      assertEquals(2, MyClass.callCount);
30    }
31  
32    public void testNoMain() throws Exception {
33      try {
34        new MainClassAdapter(MainClassAdapterTest.class, "a", "b");
35      }
36      catch (RuntimeException e) {
37        assertEquals("Class org.uispec4j.interception.MainClassAdapterTest has no method: public static void main(String[])",
38                     e.getMessage());
39      }
40    }
41  
42    public static class MyClass {
43      public static int callCount = 0;
44  
45      public static void main(String[] args) {
46        callCount++;
47        JFrame frame = new JFrame();
48        frame.getContentPane().add(new JLabel(Arrays.asList(args).toString()));
49        frame.setVisible(true);
50      }
51    }
