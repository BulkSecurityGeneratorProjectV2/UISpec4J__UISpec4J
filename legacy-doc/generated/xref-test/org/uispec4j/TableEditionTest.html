53    public void testAssertEditable() throws Exception {
54      jTable.setModel(new MyModel() {
55        public boolean isCellEditable(int row, int col) {
56          return (row + col) % 2 == 0;
57        }
58      });
59      assertTrue(table.isEditable(new boolean[][]{
60        {true, false, true},
61        {false, true, false}
62      }));
63  
64      try {
65        assertTrue(table.isEditable(new boolean[][]{
66          {false, false, false},
67          {false, false, false}
68        }));
69        throw new AssertionFailureNotDetectedError();
70      }
71      catch (AssertionError e) {
72        assertEquals("Error at row 0:\n" +
73                     "Expected: [false,false,false]\n" +
74                     "Actual:   [true,false,true]",
75                     e.getMessage());
76      }
77    }

79    public void testAssertCellEditable() throws Exception {
80      jTable.setModel(new MyModel() {
81        public boolean isCellEditable(int row, int col) {
82          return (col == 0) || ((col == 1) && (row == 1));
83        }
84      });
85  
86      checkAssertCellEditable(new boolean[][]{
87        {true, false, false},
88        {true, true, false}
89      });
90    }

92    private void checkAssertCellEditable(boolean[][] values) {
93      for (int row = 0; row < values.length; row++) {
94        boolean[] rowValues = values[row];
95        for (int column = 0; column < rowValues.length; column++) {
96          boolean value = rowValues[column];
97          Assertion result = table.cellIsEditable(row, column);
98          assertTrue((value) ? result : not(result));
99        }
100     }
101   }

149   public void testEditingACellWithAComboBox() throws Exception {
150     String[] choices = new String[]{"a", "b", "c"};
151     jTable.setDefaultEditor(String.class, new DefaultCellEditor(new JComboBox(choices)));
152     assertTrue(table.contentEquals(new Object[][]{
153       {"a", Boolean.TRUE, "3"},
154       {"c", Boolean.FALSE, "4"}
155     }));
156     ComboBox comboBox = table.editCell(0, 0).getComboBox();
157     assertTrue(comboBox.contentEquals(choices));
158     comboBox.select("b");
159     assertTrue(table.contentEquals(new Object[][]{
160       {"b", Boolean.TRUE, "3"},
161       {"c", Boolean.FALSE, "4"}
162     }));
163   }
