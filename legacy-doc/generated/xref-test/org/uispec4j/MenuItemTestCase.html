132   private void checkGetSubmenuError(String[] menus, String searchedItem, String expectedMessage) {
133     MenuBuilder builder = getBuilder("");
134     for (int i = 0; i < menus.length; i++) {
135       builder.add(menus[i]);
136     }
137     MenuItem item = builder.getMenuItem();
138     try {
139       item.getSubMenu(searchedItem);
140       throw new AssertionFailureNotDetectedError();
141     }
142     catch (AssertionError e) {
143       assertEquals(expectedMessage,
144                    e.getMessage());
145     }
146   }

160   public void testHandlingADialogShownByAPopupMenu() throws Exception {
161     final EventLogger logger = new EventLogger();
162 
163     final JPanel panel = new JPanel();
164     WindowInterceptor.run(new Trigger() {
165       public void run() throws Exception {
166         showDialog(panel);
167       }
168     });
169 
170     final JButton button = new JButton(new AbstractAction("ok") {
171       public void actionPerformed(ActionEvent e) {
172         logger.log("ok");
173       }
174     });
175 
176     MenuItem menuItem =
177       getBuilder("")
178         .add(new AbstractAction("menu") {
179           public void actionPerformed(ActionEvent e) {
180             logger.log("dialog");
181             showDialog(button);
182           }
183         })
184         .getMenuItem();
185     WindowInterceptor
186       .init(menuItem.getSubMenu("menu").triggerClick())
187       .process(new WindowHandler() {
188         public Trigger process(Window window) throws Exception {
189           logger.log("handleWindow");
190           return window.getButton("ok").triggerClick();
191         }
192       })
193       .run();
194 
195     logger.assertEquals("<log>" +
196                         "  <dialog/>" +
197                         "  <handleWindow/>" +
198                         "  <ok/>" +
199                         "</log>");
200   }
