15    public void testClickFailsIfTheMenuItemIsNotEnabled() throws Exception {
16      JPopupMenu menu = new JPopupMenu();
17      menu.add("a");
18      menu.add("b");
19      menu.add("c");
20      MenuItem item = createMenuItem(menu);
21      try {
22        item.click();
23        throw new AssertionFailureNotDetectedError();
24      }
25      catch (AssertionError e) {
26        assertEquals("This operation is not supported. You must first select a sub menu among: [a,b,c]",
27                     e.getMessage());
28      }
29    }
30  
31    protected MenuItem createLoggingMenuItem(final EventLogger eventLogger) {
32      final Action action = new AbstractAction("item") {
33        public void actionPerformed(ActionEvent e) {
34          eventLogger.log("action");
35        }
36      };
37      JPopupMenu menu = new JPopupMenu();
38      menu.add(action);
39      return createMenuItem(menu);
40    }
41  
42    static private MenuItem createMenuItem(JPopupMenu menu) {
43      return (MenuItem)UIComponentFactory.createUIComponent(menu);
44    }
