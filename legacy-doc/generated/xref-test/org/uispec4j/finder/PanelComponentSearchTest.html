494   private void checkComponentUnicityAmbiguity(Class componentClass, String uiComponentType, String componentName)
495     throws Exception {
496     Component component1 = createSwingInstance(componentClass);
497     addComponentToPanelWithName(component1, componentName + "1");
498     Component component2 = createSwingInstance(componentClass);
499     addComponentToPanelWithName(component2, componentName + "2");
500     try {
501       getAccessor(uiComponentType).getComponent();
502       fail();
503     }
504     catch (ComponentAmbiguityException e) {
505       Component[] components = new Component[]{component1, component2};
506       String[] names = new String[components.length];
507       for (int i = 0; i < components.length; i++) {
508         Component component = components[i];
509         String displayedName = ComponentUtils.getDisplayedName(component);
510         names[i] = (displayedName == null || displayedName.length() == 0) ? component.getName() : displayedName;
511       }
512       assertEquals(Messages.computeAmbiguityMessage(names, uiComponentType, null),
513                    e.getMessage());
514     }
515   }

842   private void addComponentWithLabel(Class<? extends Component> componentClass, String label) throws Exception {
843     Component component = componentClass.newInstance();
844     JLabel label1 = new JLabel(label);
