11  import java.lang.reflect.Method;

13  public class GeneratedJarChecker {
14    public static void main(String[] args) throws Exception {
15      Class componentClass = getComponentClass(args[0]);
16  
17      JCountingButton jCountingButton = new JCountingButton("counter");
18      Object newPanel = createUISpecPanel(Panel.class, jCountingButton);
19  
20      checkGetCoutingButtonMethods(newPanel, jCountingButton, componentClass);
21  
22      System.out.println("OK");
23    }
24  
25    private static void checkGetCoutingButtonMethods(Object newPanel, JCountingButton jCountingButton, Class componentClass)
26      throws NoSuchMethodException, IllegalAccessException, InvocationTargetException {
27      Method methodWithStringArg = Panel.class.getMethod("getCountingButton", String.class);
28      checkGetCountingButton(methodWithStringArg, newPanel, jCountingButton, new Object[]{"counter"}, componentClass);
29  
30      Method methodWithMatcherArg = Panel.class.getMethod("getCountingButton", ComponentMatcher.class);
31      checkGetCountingButton(methodWithMatcherArg, newPanel, jCountingButton,
32                             new Object[]{ComponentMatchers.displayedNameIdentity("counter")}, componentClass);
33  
34      Method methodWithNoArgs = Panel.class.getMethod("getCountingButton");
35      checkGetCountingButton(methodWithNoArgs, newPanel, jCountingButton, new Object[]{}, componentClass);
36    }
37  
38    private static Class getComponentClass(String classname) throws ClassNotFoundException {
39      return GeneratedJarChecker.class.getClassLoader().loadClass(classname);
40    }
41  
42    private static void checkGetCountingButton(Method method, Object newPanel, JCountingButton jCountingButton, Object[] args, Class componentClass) throws IllegalAccessException, InvocationTargetException, NoSuchMethodException {
43      Object found = method.invoke(newPanel, args);
44      check("The component should be an instance of " + componentClass.getName(), componentClass.isInstance(found));
45      jCountingButton.reset();
46      componentClass.getMethod("click").invoke(found);
47      int count = jCountingButton.getCount();
48      check("the button should have detected 1 event instead of " + count, count == 1);
49    }
50  
51    private static Object createUISpecPanel(Class newPanelClass, JComponent component) throws NoSuchMethodException, InstantiationException, IllegalAccessException, InvocationTargetException {
52      JPanel jPanel = new JPanel();
53      jPanel.add(component);
54      Constructor constructor = newPanelClass.getConstructor(Container.class);
55      return constructor.newInstance(jPanel);
56    }
57  
58    private static void check(String s, boolean b) {
59      if (!b) {
60        throw new RuntimeException(s);
61      }
62    }
63  
64  }
