11  public abstract class TableTestCase extends UnitTestCase {
12    Table table;
13    JTable jTable;
14  
15    protected void setUp() throws Exception {
16      super.setUp();
17      init(new JTable(new TableSelectionTest.MyModel()));
18    }
19  
20    private void init(JTable table) {
21      jTable = table;
22      jTable.setSelectionMode(ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
23      jTable.setName("myTable");
24      jTable.setDefaultEditor(Integer.class, new DefaultCellEditor(new JComboBox(new Object[]{3, 4, 5})));
25      this.table = (Table)UIComponentFactory.createUIComponent(jTable);
26    }
27  
28    protected class MyModel extends AbstractTableModel {
29      String[] columnNames = new String[]{"0", "1", "2"};
30      Object[][] rowData = new Object[][]{{"a", Boolean.TRUE, 3},
31                                          {"c", Boolean.FALSE, 4}};
32      Class[] columnClasses = new Class[]{String.class, Boolean.class, Integer.class};
33  
34      public String getColumnName(int column) {
35        return columnNames[column];
36      }
37  
38      public int getRowCount() {
39        return rowData.length;
40      }
41  
42      public int getColumnCount() {
43        return columnNames.length;
44      }
45  
46      public Object getValueAt(int row, int col) {
47        return rowData[row][col];
48      }
49  
50      public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
51        rowData[rowIndex][columnIndex] = aValue;
52        fireTableCellUpdated(rowIndex, columnIndex);
53      }
54  
55      public Class getColumnClass(int columnIndex) {
56        return columnClasses[columnIndex];
57      }
58  
59      public boolean isCellEditable(int row, int column) {
60        return (row != 1) || (column != 0);
61      }
62    }
63  
64    void installJPanelRenderer() {
65      DefaultTableCellRenderer renderer = new DefaultTableCellRenderer() {
66        public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
67          return new JPanel();
68        }
69      };
70      jTable.setDefaultRenderer(Boolean.class, renderer);
71    }
