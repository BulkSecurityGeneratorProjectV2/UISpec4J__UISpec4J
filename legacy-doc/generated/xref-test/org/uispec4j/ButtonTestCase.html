13  public abstract class ButtonTestCase extends UIComponentTestCase {

15    protected abstract AbstractButton getButton();

17    protected abstract javax.swing.AbstractButton getSwingButton();

19    protected UIComponent createComponent() {
20      return getButton();

22  
23    public void testEnableDisable() throws Exception {
24      checkEnabled(true);
25      checkEnabled(false);
26    }
27  
28    private void checkEnabled(boolean enabled) {
29      getSwingButton().setEnabled(enabled);
30      Assertion enabledAssertion = getButton().isEnabled();
31      assertTrue((enabled) ? enabledAssertion : not(enabledAssertion));
32    }
33  
34    public void testCheckText() throws Exception {
35      getSwingButton().setText("text");
36      assertTrue(getButton().textEquals("text"));
37      try {
38        assertTrue(getButton().textEquals("error"));
39        throw new AssertionFailureNotDetectedError();
40      }
41      catch (AssertionError e) {
42      }
43    }
44  
45    public void testCheckTextAcceptsNullText() throws Exception {
46      getSwingButton().setText(null);
47      assertTrue(getButton().textEquals(null));
48    }
49  
50    public void testCheckTextTrimsTheActualButtonText() throws Exception {
51      getSwingButton().setText(" text  ");
52      assertTrue(getButton().textEquals("text"));
53    }
54  
55    public void testIcons() throws Exception {
56      Icon icon = new Empty.DummyIcon();
57      checkAssertionFails(getButton().iconEquals(icon), "The component contains no icon.");
58  
59      getSwingButton().setIcon(icon);
60      assertTrue(getButton().iconEquals(icon));
61  
62      assertFalse(getButton().iconEquals(new Empty.DummyIcon()));
63    }
64  
65    public void testActivateIsRejectedIfTheButtonIsDisabled() throws Exception {
66      getSwingButton().setEnabled(false);
67      try {
68        getButton().click();
69        throw new AssertionFailureNotDetectedError();
70      }
71      catch (AssertionError e) {
72        assertEquals("The button is not enabled, it cannot be activated", e.getMessage());
73      }
74    }

76    public void testClick() throws Exception {
77      DummyActionListener listener = new DummyActionListener();
78      getSwingButton().addActionListener(listener);
79      getButton().click();
80      assertEquals(1, listener.getCallCount());
81    }

83    public void testTriggerClick() throws Exception {
84      DummyActionListener listener = new DummyActionListener();

86      getButton().triggerClick().run();
87      assertEquals(1, listener.getCallCount());

90    public void testClickTakesLessTimeThanWithDefaultSwingCalls() throws Exception {
91      Chrono chrono = Chrono.start();
92      getButton().click();
93      chrono.assertElapsedTimeLessThan(30);
94    }
95  
96    public void testWaitForEnabledState() throws Exception {
97      AbstractButton button = getButton();
98      final javax.swing.AbstractButton swingButton = (javax.swing.AbstractButton)button.getAwtComponent();
99      swingButton.setEnabled(false);
100     Thread thread = new Thread(new Runnable() {
101       public void run() {
102         Utils.sleep(10);
103         swingButton.setEnabled(true);
104       }
105     });
106     thread.start();
107     assertFalse(button.isEnabled());
108     UISpecAssert.waitUntil(button.isEnabled(), 30);
109     assertTrue(button.isEnabled());
110   }

114     getSwingButton().addActionListener(listener);
