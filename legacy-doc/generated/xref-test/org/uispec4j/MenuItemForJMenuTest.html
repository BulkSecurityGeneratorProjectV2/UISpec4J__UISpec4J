14    protected MenuItem createLoggingMenuItem(final EventLogger eventLogger) {
15      final Action action = new AbstractAction("item") {
16        public void actionPerformed(ActionEvent e) {
17          eventLogger.log("action");
18        }
19      };
20      JPopupMenu menu = new JPopupMenu();
21      menu.add(action);
22      return createMenuItem(menu);
23    }

25    static private MenuItem createMenuItem(JPopupMenu menu) {
