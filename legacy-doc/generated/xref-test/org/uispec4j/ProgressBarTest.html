38    public void testAssertValueEquals() throws Exception {

44    public void testCompleted() throws Exception {
45      setProgressValues(5, 15, 15);
46      assertTrue(progressBar.completionEquals(100));
47      assertTrue(progressBar.isCompleted());
48  
49      setProgressValues(0, 10, 5);
50      try {
51        assertTrue(progressBar.isCompleted());
52        throw new AssertionFailureNotDetectedError();
53      }
54      catch (AssertionError e) {
55        assertEquals("Unexpected completion rate - expected:<100> but was:<50>", e.getMessage());
56      }
57  
58      jProgressBar.setIndeterminate(true);
59      try {
60        assertTrue(progressBar.isCompleted());
61        throw new AssertionFailureNotDetectedError();
62      }
63      catch (AssertionError e) {
64        assertEquals("Unexpected completion rate - expected:<100> but was:<-1>", e.getMessage());
65      }
66    }

68    public void testAssertCompletionEqualsAcceptsValuesBetween0And100() throws Exception {
69      checkAssertCompletionError(-2, "Expected value should be in range [0,100]");
70      checkAssertCompletionError(101, "Expected value should be in range [0,100]");
71    }

73    public void testExpectedValueIsMinusOneWhenTheProgressBarIsUndeterminate() throws Exception {
74      jProgressBar.setIndeterminate(false);
75      try {
76        assertTrue(progressBar.completionEquals(-1));
77        throw new AssertionFailureNotDetectedError();

130   private void setProgressValues(int min, int max, int value) {
131     jProgressBar.setMinimum(min);
132     jProgressBar.setMaximum(max);
133     jProgressBar.setValue(value);
134   }
135 
136   private void checkAssertCompletionError(int expectedValue, String errorMessage) {
137     try {
138       assertTrue(progressBar.completionEquals(expectedValue));
139       throw new AssertionFailureNotDetectedError();
140     }
141     catch (AssertionError e) {
142       assertEquals(errorMessage, e.getMessage());
143     }
144   }
145 
146   private void checkAssertCompletionError(int expectedValue, int actualValue) {
147     checkAssertCompletionError(expectedValue,
148                                "Unexpected completion rate - expected:<" + expectedValue + "> but was:<" + actualValue + ">");
149   }
150 
151   private void checkWaitForCompletion() throws InterruptedException {
152     setProgressValues(0, 100, 0);
153     Thread thread = new Thread() {
154       public void run() {
155         Utils.sleep(50);
156         jProgressBar.setValue(100);
157         jProgressBar.setIndeterminate(false);
158         jProgressBar.setValue(100);
159       }
160     };
161     thread.start();
162     UISpecAssert.waitUntil(progressBar.isCompleted(), 200);
163     progressBar.completionEquals(100);
164     thread.join(1000);
165   }
