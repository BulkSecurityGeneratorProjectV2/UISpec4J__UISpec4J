1   package org.uispec4j.utils;

3   import java.text.DateFormat;
4   import java.text.SimpleDateFormat;
5   import java.util.Calendar;
6   import java.util.Date;
7   import java.util.Locale;
8   
9   public class DateUtilsTest extends UnitTestCase {
10  
11    private Date date = getDate(1974, Calendar.NOVEMBER, 23, 19, 55, 0);
12  
13    public void testDate() throws Exception {
14      assertEquals(date.toString(), DateUtils.getDate("1974.11.23 19:55").toString());
15    }
16  
17    public void testStandard() throws Exception {
18      assertEquals("1974.11.23 19:55", DateUtils.getStandardDate(date));
19    }
20  
21    public void testFormatted() throws Exception {
22      Locale defaultLocale = Locale.getDefault();
23      Locale.setDefault(new Locale("en", "us"));
24      assertEquals("November 23, 1974 7:55 PM", getFormattedDate(date));
25      Locale.setDefault(defaultLocale);
26    }
27  
28    public void testSetDateFormat() throws Exception {
29      Date date = getDate(2003, Calendar.DECEMBER, 25, 12, 0, 0);
30  
31      DateUtils.setDateFormat(new SimpleDateFormat("yyyy.MM.dd"));
32      assertEquals("2003.12.25", DateUtils.getStandardDate(date));
33  
34      DateUtils.setDateFormat(DateUtils.DEFAULT_DATE_FORMAT);
35      assertEquals("2003.12.25 12:00", DateUtils.getStandardDate(date));
36    }
37  
38    private static String getFormattedDate(Date date) {
39      return DateFormat.getDateTimeInstance(DateFormat.LONG, DateFormat.SHORT).format(date);
40    }
41  
42    private static Date getDate(int year, int month, int day, int hour, int minute, int second) {
43      Calendar calendar = Calendar.getInstance();
44      calendar.set(year, month, day, hour, minute, second);
45      return calendar.getTime();
46    }
47  }
