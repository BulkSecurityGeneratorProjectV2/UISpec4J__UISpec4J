308   private void checkSwitchBetweenPages(JTextComponent textComponent, String content1, String content2) {
309     TextBox textBox = new TextBox(textComponent);

320   private JTextPane createTextPaneFromUrl(URL url) throws IOException {
321     final JTextPane jTextPane = new JTextPane();
322     jTextPane.setContentType("text/html");
323     jTextPane.setPage(url);
324 
325     jTextPane.addHyperlinkListener(new HyperlinkListener() {
326       public void hyperlinkUpdate(HyperlinkEvent event) {
327         try {
328           jTextPane.setPage(event.getURL());
329         }
330         catch (IOException e) {
331         }
332       }
333     });
334     return jTextPane;
335   }

342   private void clickOnHyperLink(JEditorPane editorPane, String link, String expectedTarget) throws Exception {
343     DummyHyperlinkListener listener = new DummyHyperlinkListener();
344     editorPane.addHyperlinkListener(listener);
345     TextBox textComponent = new TextBox(editorPane);
346     textComponent.clickOnHyperlink(link);
347     assertEquals(1, listener.getCallCount());
348     assertEquals(expectedTarget, listener.getLastEvent().getDescription());
349 
350     listener.reset();
351     textComponent.triggerClickOnHyperlink(link).run();
352     assertEquals(1, listener.getCallCount());
353     assertEquals(expectedTarget, listener.getLastEvent().getDescription());
354   }

356   private void checkClickOnHyperlinkError(String html, final String link, String errorMessage) throws Exception {
357     final TextBox textComponent = new TextBox(createHtmlTextPane(html));
358     checkAssertionError(new Functor() {
359       public void run() throws Exception {
360         textComponent.clickOnHyperlink(link);
361       }
362     }, errorMessage);
363 
364     checkAssertionError(new Functor() {
365       public void run() throws Exception {
366         textComponent.triggerClickOnHyperlink(link).run();
367       }
368     }, errorMessage);
369   }

371   private static class DummyHyperlinkListener implements HyperlinkListener {
372     int callCount;
373     HyperlinkEvent lastEvent;
374 
375     public void hyperlinkUpdate(HyperlinkEvent event) {
376       callCount++;
377       this.lastEvent = event;
378     }
379 
380     public int getCallCount() {
381       return callCount;
382     }
383 
384     public HyperlinkEvent getLastEvent() {
385       return lastEvent;
386     }
387 
388     public void reset() {
389       callCount = 0;
390       lastEvent = null;
391     }
392   }
