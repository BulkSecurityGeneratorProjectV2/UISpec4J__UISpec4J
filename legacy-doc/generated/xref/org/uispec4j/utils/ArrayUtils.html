34    public static String toString(Object[][] objects) {
35      StringBuffer buffer = new StringBuffer();
36      buffer.append('[');
37      for (int i = 0; i < objects.length; i++) {
38        if (i > 0) {
39          buffer.append('\n');
40          buffer.append(' ');
41        }
42        appendLine(buffer, objects[i], ",\t");
43      }
44      buffer.append(']');
45      return buffer.toString();
46    }
47  
48    private static void appendLine(StringBuffer buffer, Object[] objects, String separator) {
49      buffer.append('[');
50      for (int i = 0; i < objects.length; i++) {
51        if (objects[i] == null) {
52          buffer.append("null");
53        }
54        else if (objects[i].getClass().isArray()) {
55          buffer.append(toString((Object[])objects[i]));
56        }
57        else {
58          buffer.append(objects[i]);
59        }
60        if (i < (objects.length - 1)) {
61          buffer.append(separator);
62        }
63      }
64      buffer.append(']');
65    }

80    public static String toString(List list) {
81      StringBuilder buffer = new StringBuilder();
82      buffer.append('[');
83      for (Iterator iterator = list.iterator(); iterator.hasNext();) {
84        buffer.append(iterator.next());
85        if (iterator.hasNext()) {
86          buffer.append(',');
87        }
88      }
89      buffer.append(']');
90      return buffer.toString();
91    }

93    public static Boolean[][] toBooleanObjects(boolean[][] source) {
94      Boolean[][] result = new Boolean[source.length][];
95      for (int i = 0; i < result.length; i++) {
96        result[i] = new Boolean[source[i].length];
97        for (int j = 0; j < result[i].length; j++) {
98          result[i][j] = source[i][j];
99        }
100     }
101     return result;
102   }

110   private static void fail(String message, Object[] expected, Object[] actual) {
111     boolean verticalDisplay = expected.length > 5 || actual.length > 5;
112     if (verticalDisplay) {
113       AssertAdapter.assertEquals(message, toVerticalString(expected), toVerticalString(actual));
114     }
115     else {
116       String prefix = message != null ? message + "\n" : "";
117       AssertAdapter.fail(prefix +
118                           "Expected: " + toString(expected) +
119                           "\nActual:   " + toString(actual));
120     }
121   }

149   public static void assertEquals(Object[] expectedArray, Iterator actualIterator) {
150     int index = 0;
151     List actualList = new ArrayList();
152     while (actualIterator.hasNext()) {
153       if (index >= expectedArray.length) {
154         for (Iterator iterator = actualIterator; iterator.hasNext();) {
155           actualList.add(iterator.next());
156         }
157         AssertAdapter.fail("The iterator contains too many elements: expected: " +
158                             toString(expectedArray) + " but was: " + actualList);
159       }
160       Object obj = actualIterator.next();
161       actualList.add(obj);
162       if (!obj.equals(expectedArray[index])) {
163         AssertAdapter.fail("Mismatch at index " + index + ". expected: " + expectedArray[index] +
164                             " but was: " + obj);
165       }
166       index++;
167     }
168     if (index < expectedArray.length) {
169       fail("Several elements are missing from the iterator", expectedArray, actualList.toArray());
170     }
171   }

177   private static void compareCollection(Object[][] actualData, Object[][] expectedData, Class collectionClass) {
178     Collection actual;
179     Collection expected;
180     try {
181       actual = (Collection)collectionClass.newInstance();
182       expected = (Collection)collectionClass.newInstance();
183     }
184     catch (InstantiationException e) {
185       throw new RuntimeException(e);
186     }
187     catch (IllegalAccessException e) {
188       throw new RuntimeException(e);
189     }
190     for (Object[] anActualData : actualData) {
191       actual.add(toString(anActualData));
192     }
193     for (Object[] anExpectedData : expectedData) {
194       expected.add(toString(anExpectedData));
195     }
196     AssertAdapter.assertEquals(expected, actual);
197   }
