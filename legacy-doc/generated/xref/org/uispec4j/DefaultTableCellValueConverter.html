4   import java.awt.*;

6   /**
7    * Default implementation for the TableCellValueConverter interface.
8    * This converter returns the displayed value for JLabel and JComboBox components, and a Boolean
9    * in the case of a JCheckBox. If another renderer type is used, it will returned the stringified object
10   * used by the internal model of the table (using toString()), or an empty string if this model
11   * object is null.
12   */
13  public class DefaultTableCellValueConverter implements TableCellValueConverter {
14  
15    public Object getValue(int row, int column, Component renderedComponent, Object modelObject) {
16      if (renderedComponent instanceof JLabel) {
17        return ((JLabel)renderedComponent).getText();
18      }
19      else if (renderedComponent instanceof JComboBox) {
20        return ((JComboBox)renderedComponent).getSelectedItem().toString();
21      }
22      else if (renderedComponent instanceof JCheckBox) {
23        return ((JCheckBox)renderedComponent).isSelected();
24      }
25      if (modelObject != null) {
26        return modelObject.toString();
27      }
28      return "";
29    }
30  }
