12  class PanelClassEnhancer extends ClassAdapter implements Opcodes {
13    private Extension[] extensions;
14  
15    public static byte[] transformClass(ClassReader reader, Extension[] extensions) {
16      ClassWriter writer = new ClassWriter(ClassWriter.COMPUTE_MAXS | ClassWriter.COMPUTE_FRAMES);
17      PanelClassEnhancer transformer = new PanelClassEnhancer(writer, extensions);
18      reader.accept(transformer, 0);
19      return writer.toByteArray();
20    }
21  
22    private PanelClassEnhancer(ClassVisitor cv, Extension[] extensions) {
23      super(cv);
24      this.extensions = extensions;
25    }
26  
27    public MethodVisitor visitMethod(int access,
28                                     String name, String desc,
29                                     String signature,
30                                     String[] exceptions) {
31  
32      MethodVisitor cd = cv.visitMethod(access, name, desc, signature, exceptions);
33      if (cd == null) {
34        return null;
35      }
36      if ("<clinit>".equals(name) && ((access & ACC_STATIC) != 0)) {
37        return new StaticInitEnhancer(cd);
38      }
39      return cd;
40    }
41  
42    private class StaticInitEnhancer extends MethodAdapter {
43      public StaticInitEnhancer(MethodVisitor mv) {
44        super(mv);
45      }
46  
47      public void visitInsn(int opcode) {
48        if (opcode == RETURN) {
49          for (Extension extension : extensions) {
50            String componentClassName = extension.getComponentClassName();
51            String transformedComponentClassName = transformClassName(componentClassName);
52            addStaticInitializerForExtension(transformedComponentClassName, componentClassName);
53          }
54        }
55        mv.visitInsn(opcode);
56      }
57  
58      private void addStaticInitializerForExtension(String transformedComponentClassName, String componentClassName) {
59        mv.visitFieldInsn(GETSTATIC, Type.getInternalName(Panel.class), transformedComponentClassName, Type.getDescriptor(Class.class));
60        Label l12 = new Label();
61        mv.visitJumpInsn(IFNONNULL, l12);
62        mv.visitLdcInsn(componentClassName);
63        mv.visitMethodInsn(INVOKESTATIC, Type.getInternalName(Panel.class), "getClass",
64                           Type.getMethodDescriptor(Type.getType(Class.class), new Type[]{Type.getType(String.class)}));
65        mv.visitInsn(DUP);
66        mv.visitFieldInsn(PUTSTATIC, Type.getInternalName(Panel.class), transformedComponentClassName, Type.getDescriptor(Class.class));
67        Label l13 = new Label();
68        mv.visitJumpInsn(GOTO, l13);
69        mv.visitLabel(l12);
70        mv.visitFieldInsn(GETSTATIC, Type.getInternalName(Panel.class), transformedComponentClassName, Type.getDescriptor(Class.class));
71        mv.visitLabel(l13);
72        mv.visitMethodInsn(INVOKESTATIC, Type.getInternalName(UIComponentFactory.class), "addUIComponentClass",
73                           Type.getMethodDescriptor(Type.VOID_TYPE, new Type[]{Type.getType(Class.class)}));
74      }
75    }
76  
77    public void visitEnd() {
78      addStaticGetClassMethod();
79      for (Extension extension : extensions) {
80        addExtension(extension.getComponentName(), extension.getComponentClassName());
81      }
82      cv.visitEnd();
83    }

85    private void addStaticGetClassMethod() {
86      MethodVisitor v = cv.visitMethod(ACC_STATIC + ACC_PRIVATE, "getClass",
87                                       Type.getMethodDescriptor(Type.getType(Class.class),
88                                                                new Type[]{Type.getType(String.class)}), null, null);
89      Label l0 = new Label();
90      v.visitLabel(l0);
91      v.visitVarInsn(ALOAD, 0);
92      v.visitMethodInsn(INVOKESTATIC, Type.getInternalName(Class.class),
93                        "forName", Type.getMethodDescriptor(Type.getType(Class.class), new Type[]{Type.getType(String.class)}));
94      Label l1 = new Label();
95      v.visitLabel(l1);
96      v.visitInsn(ARETURN);
97      Label l2 = new Label();
98      v.visitLabel(l2);
99      v.visitVarInsn(ASTORE, 1);
100     v.visitTypeInsn(NEW, Type.getInternalName(NoClassDefFoundError.class));
101     v.visitInsn(DUP);
102     v.visitVarInsn(ALOAD, 1);
103     v.visitMethodInsn(INVOKEVIRTUAL, Type.getInternalName(Throwable.class), "getMessage",
104                       Type.getMethodDescriptor(Type.getType(String.class), new Type[0]));
105     v.visitMethodInsn(INVOKESPECIAL, Type.getInternalName(NoClassDefFoundError.class), "<init>",
106                       Type.getMethodDescriptor(Type.VOID_TYPE, new Type[]{Type.getType(String.class)}));
107     v.visitInsn(ATHROW);
108     v.visitTryCatchBlock(l0, l1, l2, Type.getInternalName(ClassNotFoundException.class));
109     v.visitMaxs(3, 2);
110   }
