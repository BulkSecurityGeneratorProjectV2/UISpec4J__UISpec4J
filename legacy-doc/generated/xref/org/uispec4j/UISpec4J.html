6   import java.awt.*;

9   /**
10   * Facade for the initialization of the library, mainly used for the interception mechanism.
11   *
12   * @see <a href="http://www.uispec4j.org/intercepting-windows">Intercepting windows</a>
13   */
14  public class UISpec4J {
15    private static long windowInterceptionTimeLimit = 10000;
16    public static final int DEFAULT_ASSERTION_TIME_LIMIT = 500;
17    private static long assertionTimeLimit = DEFAULT_ASSERTION_TIME_LIMIT;
18  
19    /**
20     * Initializes UISpec4J, for instance by setting up the interception mechanism.
21     */
22    public static void init() {
23      if (GraphicsEnvironment.isHeadless()) {
24        System.out.println("UISpec4J.init: warning: headless mode is not supported");
25      }
26  
27      // Black magic - do not touch this (system.setProperty seem to load dynamic libraries)
28      if ("Linux".equalsIgnoreCase(System.getProperty("os.name")) && "1.5".equals(System.getProperty("java.specification.version"))) {
29        System.setProperty("awt.toolkit", "sun.awt.motif.MToolkit");
30      }
31      UISpecLF.init();
32      initToolkit();
33    }
34  
35    private static void initToolkit() {
36      try {
37        Field toolkitField = Toolkit.class.getDeclaredField("toolkit");
38        toolkitField.setAccessible(true);
39        toolkitField.set(null, new UISpecToolkit());
40      }
41      catch (Exception e) {
42        throw new RuntimeException("Unable to initialize toolkit for interception.", e);
43      }
44    }
45  
46    /**
47     * Sets the default timeout value (in milliseconds) used by the library for interception mechanism.
48     * Default value is 10.000, i.e. 10s.
49     */
50    public static void setWindowInterceptionTimeLimit(long millis) {
51      UISpec4J.windowInterceptionTimeLimit = millis;
52    }
53  
54    /**
55     * Returns the default timeout value (in milliseconds) used by the library for interception mechanism.
56     * Default value is 10.000, i.e. 10s.
57     */
58    public static long getWindowInterceptionTimeLimit() {
59      return windowInterceptionTimeLimit;
60    }
61  
62    /**
63     * Returns the default timeout value (in milliseconds) used when checking assertions.
64     * Default value is 500, i.e. 0.5s.
65     */
66    public static long getAssertionTimeLimit() {
67      return assertionTimeLimit;
68    }
69  
70    /**
71     * Sets the default timeout value (in milliseconds) used when checking assertions.
72     * Default value is 500, i.e. 0.5s.
73     */
74    public static void setAssertionTimeLimit(long assertionTimeLimit) {
75      UISpec4J.assertionTimeLimit = assertionTimeLimit;
76    }
77  }
