11  import javax.swing.*;
12  
13  /**
14   * Interceptor for (usually right-click triggered) pop-up menus.<p>
15   * For instance, in the following code snippedt we right-click in the (2,4) cell of a table,
16   * then click on the "Copy" item found in the popped-up menu:
17   * <pre><code>
18   * PopupMenuInterceptor
19   *   .run(table.triggerRightClick(2, 4))
20   *   .getSubMenu("Copy")
21   *   .click();
22   * </code></pre>
23   *
24   * @see <a href="http://www.uispec4j.org/intercepting-windows">Intercepting windows</a>
25   */
26  public class PopupMenuInterceptor {
27  
28    /**
29     * <p>Runs the given trigger and returns the intercepted popup menu.</p>
30     * This method will wait for the popup to be shown, up to a time specified with
31     * {@link UISpec4J#setWindowInterceptionTimeLimit(long)}.
32     *
33     * @throws Error if no popup is shown before the timeout expires.
34     */
35    public static MenuItem run(final Trigger trigger) {
36      PopupHandler interceptor = new PopupHandler();
37      final UISpecDisplay display = UISpecDisplay.instance();
38      try {
39        display.add(interceptor);
40        display.setCurrentPopup(null);
41        TriggerRunner.runInSwingThread(trigger);
42        UISpecAssert.waitUntil("No popup was shown",
43                               new Assertion() {
44                                 public void check() {
45                                   if (display.getCurrentPopup() == null) {
46                                     AssertAdapter.fail("No popup shown");
47                                   }
48                                   ;
49                                 }
50                               },
51                               UISpec4J.getWindowInterceptionTimeLimit());
52        return new MenuItem(display.getCurrentPopup());
53      }
54      finally {
55        display.remove(interceptor);
56      }
57    }
58  
59    private static class PopupHandler implements InterceptionHandler {
60      MenuItem menu;
61  
62      public void process(Window window) {
63        try {
64          this.menu = new MenuItem((JPopupMenu)window.findSwingComponent(JPopupMenu.class));
65        }
66        catch (ItemNotFoundException e) {
67        }
68        catch (ComponentAmbiguityException e) {
69        }
70      }
71    }
72  }
