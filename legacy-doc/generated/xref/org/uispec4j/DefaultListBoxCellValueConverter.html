4   import java.awt.*;

6   /**
7    * Default implementation for the ListBoxCellValueConverter interface.
8    * Returns the text displayed by the renderer, only in the case of JLabel renderers -
9    * this method will return an empty String if the renderer is not a JLabel.
10   */
11  public class DefaultListBoxCellValueConverter implements ListBoxCellValueConverter {
12    public String getValue(int index, Component renderedComponent, Object modelObject) {
13      if (renderedComponent instanceof JLabel) {
14        return ((JLabel)renderedComponent).getText();
15      }
16      if (modelObject != null) {
17        return modelObject.toString();
18      }
19      return "";
20    }
21  }
