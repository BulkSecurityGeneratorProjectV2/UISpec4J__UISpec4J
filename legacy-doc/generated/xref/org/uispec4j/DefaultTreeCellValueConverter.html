3   import javax.swing.*;
4   import java.awt.*;

6   /**
7    * Default implementation for the TreeCellValueConverter interface.
8    * This converter returns the text displayed by JLabel components only - it will throw an
9    * exception if another renderer component is not used.
10   */
11  public class DefaultTreeCellValueConverter implements TreeCellValueConverter {
12    public String getValue(Component renderedComponent, Object modelObject) {
13      return getLabel(renderedComponent).getText();
14    }
15  
16    public boolean isBold(Component renderedComponent, Object modelObject) {
17      Font font = getLabel(renderedComponent).getFont();
18      return (font != null) && font.isBold();
19    }
20  
21    public Color getForeground(Component renderedComponent, Object modelObject) {
22      return getLabel(renderedComponent).getForeground();
23    }
24  
25    protected JLabel getLabel(Component renderedComponent) {
26      if (!JLabel.class.isInstance(renderedComponent)) {
27        throw new RuntimeException("Unexpected renderer for jTree: only JLabel is supported");
28      }
29      return (JLabel)renderedComponent;
30    }
31  }
