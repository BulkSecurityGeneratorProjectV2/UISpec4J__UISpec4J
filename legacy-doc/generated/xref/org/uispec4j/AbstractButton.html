3   import org.uispec4j.assertion.Assertion;
4   import org.uispec4j.assertion.testlibrairies.AssertAdapter;
5   import org.uispec4j.utils.TriggerRunner;
6   
7   import javax.swing.*;
8   
9   /**
10   * Base class for button-like components (toggle buttons, check boxes, etc.)
11   */
12  public abstract class AbstractButton extends AbstractSwingUIComponent {
13    private javax.swing.AbstractButton abstractButton;
14  
15    protected AbstractButton(javax.swing.AbstractButton abstractButton) {
16      this.abstractButton = abstractButton;
17    }
18  
19    public void click() {
20      AssertAdapter.assertTrue("The button is not enabled, it cannot be activated",
21                               abstractButton.isEnabled());
22      AssertAdapter.assertTrue("The button is not visible, it cannot be activated",
23                               abstractButton.isVisible());
24  
25      if (SwingUtilities.isEventDispatchThread()) {
26        doClick(abstractButton);
27      }
28      else {
29        TriggerRunner.runInSwingThread(triggerClick());
30      }
31    }
32  
33    static void doClick(javax.swing.AbstractButton button) {
34      ButtonModel model = button.getModel();
35      model.setArmed(true);
36      model.setPressed(true);
37      model.setPressed(false);
38      model.setArmed(false);
39    }
40  
41    public Assertion textEquals(final String text) {
42      return new Assertion() {
43        public void check() {
44          String label = abstractButton.getText();
45          if (label != null) {
46            label = label.trim();
47          }
48          AssertAdapter.assertEquals(text, label);
49        }
50      };
51    }
52  
53    /**
54     * Checks the icon displayed by the component. Please note that equals()
55     * not being defined for Icon implementations, you will have to provide a pointer
56     * to the actual Icon instance that is used in the production code. This make
57     * this method mostly suited to unit testing.
58     */
59    public Assertion iconEquals(final Icon expected) {
60      return new Assertion() {
61        public void check() {
62          Icon actual = abstractButton.getIcon();
63          if (expected != null) {
64            AssertAdapter.assertNotNull("The component contains no icon.", actual);
65          }
66          AssertAdapter.assertSame("The icon ", expected, actual);
67        }
68      };
69    }
70  
71    public String getLabel() {
72      return abstractButton.getText();
73    }
74  
75    public Trigger triggerClick() {
76      return new Trigger() {
77        public void run() {
78          AbstractButton.this.click();
79        }
80      };
81    }
82  }
